-- YARDIMCI FONKSİYON: WAVE GRADIENT OLUŞTURUCU
local function ApplyWaveGradient(parent, color1, color2)
    parent.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    local grad = Instance.new("UIGradient", parent)
    grad.Color = ColorSequence.new({
        ColorSequenceKeypoint.new(0, color1),
        ColorSequenceKeypoint.new(1, color2)
    })
    grad.Rotation = 45
    return grad
end

local dragging, dragInput, dragStart, startPos
MainFrame.InputBegan:Connect(function(input)
    if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) and not MainFrame:GetAttribute("Minimized") then
        dragging = true; dragStart = input.Position; startPos = MainFrame.Position
        input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
    end
end)
MainFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end
end)
UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

local lastTPTarget = nil
local function GetPlayerFromPart(str)
    str = string.lower(str)
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and (string.find(string.lower(p.Name), str) or string.find(string.lower(p.DisplayName), str)) then return p end
    end
    return nil
end
local function ExecuteTP(targetPlayer)
    if not targetPlayer or not targetPlayer.Character or not targetPlayer.Character:FindFirstChild("HumanoidRootPart") then
        ShowNotification("Teleport Error", Translates[CurrentLanguage].TpError, 3); return
    end
    local myChar = LocalPlayer.Character
    if myChar and myChar:FindFirstChild("HumanoidRootPart") then
        myChar.HumanoidRootPart.CFrame = targetPlayer.Character.HumanoidRootPart.CFrame
        lastTPTarget = targetPlayer
        ShowNotification("Teleport", Translates[CurrentLanguage].TpSuccess .. targetPlayer.Name, 3)
    end
end

TpBtn.MouseButton1Click:Connect(function() ExecuteTP(GetPlayerFromPart(TpBox.Text)) end)
TpLastBtn.MouseButton1Click:Connect(function() ExecuteTP(lastTPTarget) end)
TpRandomBtn.MouseButton1Click:Connect(function()
    local plrs, valid = Players:GetPlayers(), {}
    for _, p in ipairs(plrs) do if p ~= LocalPlayer then table.insert(valid, p) end end
    if #valid > 0 then ExecuteTP(valid[math.random(1, #valid)]) else ShowNotification("Teleport Error", Translates[CurrentLanguage].NoOtherPlrs, 3) end
end)

        local cam = workspace.CurrentCamera; flyBG.CFrame = cam.CFrame; local moveVec = Vector3.new(0,0,0)
        pcall(function() moveVec = require(LocalPlayer.PlayerScripts:WaitForChild("PlayerModule")):GetControls():GetMoveVector() end)
        local moveDir = (cam.CFrame.RightVector * moveVec.X) + (cam.CFrame.LookVector * -moveVec.Z)
        if moveDir.Magnitude > 0 then moveDir = moveDir.Unit end; flyBV.Velocity = moveDir * flySpeed
    end)
    ShowNotification("Fly", Translates[CurrentLanguage].FlyActive, 3)
end
local function StopFly()
    isFlying = false
    if flyBV then flyBV:Destroy() end; if flyBG then flyBG:Destroy() end; if flyLoop then flyLoop:Disconnect() end
    if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then LocalPlayer.Character.Humanoid.PlatformStand = false end
    ShowNotification("Fly", Translates[CurrentLanguage].FlyDeactive, 2)
end

local lastTPTarget = nil
local function GetPlayerFromPart(str)
    str = string.lower(str)
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and (string.find(string.lower(p.Name), str) or string.find(string.lower(p.DisplayName), str)) then return p end
    end
    return nil
end
local function ExecuteTP(targetPlayer)
    if not targetPlayer or not targetPlayer.Character or not targetPlayer.Character:FindFirstChild("HumanoidRootPart") then
        ShowNotification("Teleport Error", Translates[CurrentLanguage].TpError, 3); return
    end
    local myChar = LocalPlayer.Character
    if myChar and myChar:FindFirstChild("HumanoidRootPart") then
        myChar.HumanoidRootPart.CFrame = targetPlayer.Character.HumanoidRootPart.CFrame
        lastTPTarget = targetPlayer
        ShowNotification("Teleport", Translates[CurrentLanguage].TpSuccess .. targetPlayer.Name, 3)
    end
end

TpBtn.MouseButton1Click:Connect(function() ExecuteTP(GetPlayerFromPart(TpBox.Text)) end)
TpLastBtn.MouseButton1Click:Connect(function() ExecuteTP(lastTPTarget) end)
TpRandomBtn.MouseButton1Click:Connect(function()
    local plrs, valid = Players:GetPlayers(), {}
    for _, p in ipairs(plrs) do if p ~= LocalPlayer then table.insert(valid, p) end end
    if #valid > 0 then ExecuteTP(valid[math.random(1, #valid)]) else ShowNotification("Teleport Error", Translates[CurrentLanguage].NoOtherPlrs, 3) end
end)

    task.spawn(function()
        local onlineData = {}
        pcall(function()
            local friendsOnline = LocalPlayer:GetFriendsOnline(200) or {}
            for _, f in ipairs(friendsOnline) do 
                onlineData[f.VisitorId] = f 
            end
        end)
        
        local success, friendPages = pcall(function() return Players:GetFriendsAsync(LocalPlayer.UserId) end)
        if not success then return end
        
        local rawFriendsList = {}
        
        while true do
            for _, friend in ipairs(friendPages:GetCurrentPage()) do
                local fId = friend.Id
                local username = friend.Username
                local displayName = friend.DisplayName or username
                
                local isOnline = false
                local isGame = false
                local isStudio = false
                local gameName = ""
                local placeId = 0
                
                if onlineData[fId] then
                    local rawLocationType = onlineData[fId].LocationType
                    gameName = onlineData[fId].LastLocation or ""
                    placeId = onlineData[fId].PlaceId or 0
                    
                    local locVal = -1
                    if typeof(rawLocationType) == "EnumItem" then
                        locVal = rawLocationType.Value
                    elseif type(rawLocationType) == "number" then
                        locVal = rawLocationType
                    end
                    
                    local isStudioLocation = string.find(string.lower(gameName), "studio") or string.find(string.lower(gameName), "editing")
                    local isWebsite = (gameName == "Website" or gameName == "Mobile Website")
                    
                    if locVal == 3 or isStudioLocation then
                        isStudio = true
                    elseif (locVal == 2 or placeId > 0) and not isStudioLocation and not isWebsite then
                        isGame = true
                    else
                        isOnline = true
                    end
                end
                
                table.insert(rawFriendsList, {
                    Id = fId,
                    Username = username,
                    DisplayName = displayName,
                    IsOnline = isOnline,
                    IsGame = isGame,
                    IsStudio = isStudio,
                    GameName = gameName,
                    PlaceId = placeId
                })
            end
            if friendPages.IsFinished then break end
            pcall(function() friendPages:AdvanceToNextPageAsync() end)
        end
        
        table.sort(rawFriendsList, function(a, b)
            local aScore = a.IsGame and 4 or (a.IsStudio and 3 or (a.IsOnline and 2 or 1))
            local bScore = b.IsGame and 4 or (b.IsStudio and 3 or (b.IsOnline and 2 or 1))
            if aScore ~= bScore then
                return aScore > bScore
            else
                return a.Username:lower() < b.Username:lower()
            end
        end)
        
        for idx, friend in ipairs(rawFriendsList) do

local currentVisiblePage = nil

local function SwitchToPage(pageName)
    local targetBtn = TabButtons[pageName]
    if TabButtons.Selected == targetBtn then return end
    
    if TabButtons.Selected then
        TweenService:Create(TabButtons.Selected, TweenInfo.new(0.2), {ImageColor3 = Color3.fromRGB(150, 151, 160)}):Play()
        local prevContainer = TabButtons.Selected.Parent
        local prevStroke = prevContainer:FindFirstChildOfClass("UIStroke")
        if prevStroke then TweenService:Create(prevStroke, TweenInfo.new(0.2), {Transparency = 1}):Play() end
        TweenService:Create(prevContainer, TweenInfo.new(0.2), {BackgroundTransparency = 1}):Play()
    end
    
    TabButtons.Selected = targetBtn
    TweenService:Create(targetBtn, TweenInfo.new(0.2), {ImageColor3 = Color3.fromRGB(255, 255, 255)}):Play()
    local currContainer = targetBtn.Parent
    local currStroke = currContainer:FindFirstChildOfClass("UIStroke")
    if currStroke then TweenService:Create(currStroke, TweenInfo.new(0.2), {Transparency = 0.9, Color = Color3.fromRGB(255, 255, 255)}):Play() end
    TweenService:Create(currContainer, TweenInfo.new(0.2), {BackgroundTransparency = 0.9, BackgroundColor3 = Color3.fromRGB(255, 255, 255)}):Play()
    
    local oldPageName = currentVisiblePage
    local newPage = Pages[pageName]
    
    if oldPageName and Pages[oldPageName] then
        local oldPage = Pages[oldPageName]
        local outTween = TweenService:Create(oldPage, TweenInfo.new(0.25, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Position = UDim2.new(1, 0, 0, 0)
        })
        outTween:Play()
        task.spawn(function()
            outTween.Completed:Wait()
            if currentVisiblePage ~= oldPageName then
                oldPage.Visible = false
            end
        end)
    end
    
    for name, page in pairs(Pages) do
        if name ~= pageName and name ~= oldPageName then
            page.Visible = false
            page.Position = UDim2.new(1, 0, 0, 0)
        end
    end
    
    currentVisiblePage = pageName
    newPage.Position = UDim2.new(-1, 0, 0, 0)
    newPage.Visible = true
    
    TweenService:Create(newPage, TweenInfo.new(0.35, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
        Position = UDim2.new(0, 0, 0, 0)
    }):Play()
end

local function PopulateFriendsList()
    for _, child in ipairs(FriendsScroll:GetChildren()) do 
        if child:IsA("Frame") then child:Destroy() end 
    end
    
    task.spawn(function()
        local onlineData = {}
        pcall(function()
            local friendsOnline = LocalPlayer:GetFriendsOnline(200) or {}
            for _, f in ipairs(friendsOnline) do 
                onlineData[f.VisitorId] = f 
            end
        end)
        
        local success, friendPages = pcall(function() return Players:GetFriendsAsync(LocalPlayer.UserId) end)
        if not success then return end
        
        local rawFriendsList = {}
        
        while true do
            for _, friend in ipairs(friendPages:GetCurrentPage()) do
                local fId = friend.Id
                local username = friend.Username
                local displayName = friend.DisplayName or username
                
                local isOnline = false
                local isGame = false
                local isStudio = false
                local gameName = ""
                local placeId = 0
                
                if onlineData[fId] then
                    local rawLocationType = onlineData[fId].LocationType
                    gameName = onlineData[fId].LastLocation or ""
                    placeId = onlineData[fId].PlaceId or 0
                    
                    local locVal = -1
                    if typeof(rawLocationType) == "EnumItem" then
                        locVal = rawLocationType.Value
                    elseif type(rawLocationType) == "number" then
                        locVal = rawLocationType
                    end
                    
                    local isStudioLocation = string.find(string.lower(gameName), "studio") or string.find(string.lower(gameName), "editing")
                    local isWebsite = (gameName == "Website" or gameName == "Mobile Website")
                    
                    if locVal == 3 or isStudioLocation then
                        isStudio = true
                    elseif (locVal == 2 or placeId > 0) and not isStudioLocation and not isWebsite then
                        isGame = true
                    else
                        isOnline = true
                    end
                end
                
                table.insert(rawFriendsList, {
                    Id = fId,

local currentVisiblePage = nil

local function SwitchToPage(pageName)
    local targetBtn = TabButtons[pageName]
    if TabButtons.Selected == targetBtn then return end
    
    if TabButtons.Selected then
        TweenService:Create(TabButtons.Selected, TweenInfo.new(0.2), {ImageColor3 = Color3.fromRGB(150, 151, 160)}):Play()
        local prevContainer = TabButtons.Selected.Parent
        local prevStroke = prevContainer:FindFirstChildOfClass("UIStroke")
        if prevStroke then TweenService:Create(prevStroke, TweenInfo.new(0.2), {Transparency = 1}):Play() end
        TweenService:Create(prevContainer, TweenInfo.new(0.2), {BackgroundTransparency = 1}):Play()
    end
    
    TabButtons.Selected = targetBtn
    TweenService:Create(targetBtn, TweenInfo.new(0.2), {ImageColor3 = Color3.fromRGB(255, 255, 255)}):Play()
    local currContainer = targetBtn.Parent
    local currStroke = currContainer:FindFirstChildOfClass("UIStroke")
    if currStroke then TweenService:Create(currStroke, TweenInfo.new(0.2), {Transparency = 0.9, Color = Color3.fromRGB(255, 255, 255)}):Play() end
    TweenService:Create(currContainer, TweenInfo.new(0.2), {BackgroundTransparency = 0.9, BackgroundColor3 = Color3.fromRGB(255, 255, 255)}):Play()
    
    local oldPageName = currentVisiblePage
    local newPage = Pages[pageName]
    
    if oldPageName and Pages[oldPageName] then
        local oldPage = Pages[oldPageName]
        local outTween = TweenService:Create(oldPage, TweenInfo.new(0.25, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Position = UDim2.new(1, 0, 0, 0)
        })
        outTween:Play()
        task.spawn(function()
            outTween.Completed:Wait()
            if currentVisiblePage ~= oldPageName then
                oldPage.Visible = false
            end
        end)
    end
    
    for name, page in pairs(Pages) do
        if name ~= pageName and name ~= oldPageName then
            page.Visible = false
            page.Position = UDim2.new(1, 0, 0, 0)
        end
    end
    
    currentVisiblePage = pageName
    newPage.Position = UDim2.new(-1, 0, 0, 0)
    newPage.Visible = true
    
    TweenService:Create(newPage, TweenInfo.new(0.35, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
        Position = UDim2.new(0, 0, 0, 0)
    }):Play()
end
